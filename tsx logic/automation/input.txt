  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Fire/inferno color pulse
  const firePulse = Math.sin(frame * 0.18) * 0.5 + 0.5;
  const fireScale = 1 + Math.sin(frame * 0.12) * 0.04;

  const shockwaveProgress2 = interpolate(frame, [20, 80], [0, 1], { extrapolateRight: 'clamp' });
  const shockwaveProgress3 = interpolate(frame, [40, 100], [0, 1], { extrapolateRight: 'clamp' });

  // Word-by-word text reveal
  const wordReveal = (wordIndex: number) =>
    interpolate(frame, [wordIndex * 4, wordIndex * 4 + 12], [0, 1], {
      extrapolateLeft: 'clamp', extrapolateRight: 'clamp', easing: Easing.out(Easing.cubic),
    });

  // Chomping animation for "frisst"

  const shakeX = frame < 20 ? Math.sin(frame * 2.7) * interpolate(frame, [0, 20], [8, 0], { extrapolateRight: 'clamp' }) : 0;
  const shakeY = frame < 20 ? Math.cos(frame * 3.1) * interpolate(frame, [0, 20], [5, 0], { extrapolateRight: 'clamp' }) : 0;
