  height: 1080,

    [durationInFrames - 15, durationInFrames],
    [1, 0],

  // Heartbeat pulse effect
  const heartbeat = Math.sin(frame * 0.18) > 0.7
    ? interpolate(Math.sin(frame * 0.18), [0.7, 1], [1, 1.15])
    : 1;

  // Slow radial breathe
  const breathe = 1 + Math.sin(frame * 0.04) * 0.025;

  // Layered reveal
  const titleSlide = sp(10);
  const subSlide = sp(28);
  const tagSlide = sp(44);

  // Scan line sweep
  const scanX = interpolate(frame, [0, durationInFrames], [-100, 2100], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  return (
    <AbsoluteFill
      style={{
        backgroundColor: '#1C1C1C',
        overflow: 'hidden',
        opacity: exitOp,
      }}
    >

            <line
              key={`h${i}`}
              x1="0"
              y1={i * 52}
              x2="1920"
              y2={i * 52}
              stroke="#e9d700"
              strokeWidth="0.5"
            />
          ))}
          {Array.from({ length: 32 }).map((_, i) => (
            <line
              key={`v${i}`}
              x1={i * 62}
              y1="0"
              x2={i * 62}
              y2="1080"
              stroke="#e9d700"
              strokeWidth="0.5"
            />
          ))}
        </svg>
