    ? Math.sin(frame * 2.5) * 8
    : 0;
  const shakeY = frame > 15 && frame < 22
    ? Math.cos(frame * 3) * 5
    : 0;

  // Pulsing red glow
  const redPulse = interpolate(
    Math.sin(frame * 0.12),
    [-1, 1],
    [0.4, 1]
  );

  // Floating particles
  const particles = Array.from({ length: 18 }, (_, i) => ({
    x: seededRandom(i * 7) * 1920,
    y: seededRandom(i * 13) * 1080,

      style={{
        backgroundColor: '#1C1C1C',

        {[380, 300, 220].map((r, i) => (
          <div
            key={i}
            style={{
              position: 'absolute',
              width: r * 2,
              height: r * 2,

              border: `2px solid #ff0000`,

                Math.sin(frame * 0.05 + i * 1.2),

              transform: `scale(${1 + Math.sin(frame * 0.04 + i) * 0.02})`,
            }}
          />
        ))}
      </AbsoluteFill>

      {/* Shape layers — large transparent triangles */}
      <svg
        style={{ position: 'absolute', width: '100%', height: '100%', opacity: 0.06 }}
        viewBox="0 0 1920 1080"
      >
        <polygon
          points="1700,50 1920,400 1480,400"
          fill="#ff0000"
          style={{
            transform: `rotate(${frame * 0.02}deg)`,
            transformOrigin: '1700px 250px',
          }}
        />
        <polygon
          points="100,700 350,1080 -150,1080"
          fill="#ff0000"
          style={{
            transform: `rotate(${-frame * 0.015}deg)`,
            transformOrigin: '100px 890px',
          }}
        />
        <rect
          x="800"
          y="-60"
          width="320"
          height="320"
